\newcommand{\statementtype}{proposition}
\newcommand{\statementoption}{Characterisation of CNPA}
\newcommand{\statementlabel}{prop:cnpa_characterisation}
\newcommand{\statementcontent}{
    Assume that $\inf_{\Qc\ss{G,MF}} \L\ss{NP} < \infty$.
    Also assume that $\tI$ is dense in $I_N$ for some $N \ge 1$.
    Then a noisy prediction map $(\pi, \sigma) \in \Qc\ss{G,MF}$ is a CNPA if and only if
    \begin{equation}
        m_\pi|_{\tD} = m_f|_{\tD}
        \quad
        \text{and}
        \quad
        v_\pi|_{\tD} + \sigma^2 = v_f|_{\tD} + \sigma_f^2,
    \end{equation}
}
