\newcommand{\statementtype}{proposition}
\newcommand{\statementoption}{Consistency of CNPA}
\newcommand{\statementlabel}{prop:cnpa_consistency}
\newcommand{\statementcontent}{
    Assume that $\tI$ is dense in $I_N$ for some $N \ge 1$,
    and assume that $\sup_{\vx \in \tI}\, \abs{\vx} < \infty$.
    Let $(\pi_M, \sigma_M) \in \widetilde \Qc\ss{G,MF}$
    be such that
    \begin{equation}
        \L_M(\pi_M, \sigma_M) \le {\textstyle\inf_{\widetilde \Qc\ss{G,MF}}}\, \L_M + o_\P(1).
    \end{equation}
    Then, as $M \to \infty$, the distance of $(\pi_M, \sigma_M)$ to the closest CNPA converges to zero in probability.
}
