\newcommand{\statementtype}{proposition}
\newcommand{\statementoption}{Characterisation of GNPA}
\newcommand{\statementlabel}{prop:gnpa_characterisation}
\newcommand{\statementcontent}{
    Assume that $\inf_{\Qc\ss{G}} \L\ss{NP} < \infty$.
    Also assume that $\tI$ is dense in $I_N$ for some $N \ge 2$.
    Then a noisy prediction map $(\pi, \sigma) \in \Qc\ss{G}$ is a GNPA if and only if 
    \begin{equation}
        m_\pi|_{\tD} = m_f|_{\tD},
        \quad
        k_\pi|_{\tD} = k_f|_{\tD},
        \quad
        \text{and}
        \quad
        \sigma = \sigma_f.
    \end{equation}
}
