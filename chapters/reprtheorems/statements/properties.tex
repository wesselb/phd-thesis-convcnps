\newcommand{\statementlabel}{prop:properties_metric_data_sets}
\newcommand{\statementcontent}{
    ~\begin{proplist}
        \item
            The function $d_{[\D]}$ is a metric on $[\D]$.
        \item
            \label{prop:properties_metric_data_sets:open_implies_open}
            If $A \sub \D$ is open, then $[A]$ is open in the topology of $d_{[\D]}$.
        \item
            \label{prop:properties_metric_data_sets:closed_implies_closed}
            If $A \sub \D$ is closed and permutation invariant, then $[A]$ is closed in the topology of $d_{[\D]}$.
        \item
            \label{prop:properties_metric_data_sets:same_topology}
            The topology on $[\D]$ induced by $d_{[\D]}$ coincides with the quotient topology.
    \end{proplist}
}
